### js中的栈和队列

##### 1.队列:

```js
// 1.在js中 队列就是数组（Array） 
// 2.队列顾名思义就是排队，遵循先进先出的规则
// 3.实现队列的需要使用到的数组方法：
	// push() 进队从数组末尾添加
	// shfit() 出队从数组的第一个删除
// 4.我们需要实现队列常见的基本方法：
	// enqueue 入队，添加到队列尾部；
	// dequeue 出队，队列最前面的数据移除；
	// peek 获取队列最前面的数据（仅获取查看，不改变队列数据）
	// size 返回队列元素的个数
	// isEmpty 返回队列是否为空
	// clear 清除队列
	// toArray 将queue转换成数组

// 5.使用 ES6的新数据格式 WeakMap(WeakMap只接受对象作为键名（null除外），不接受其他类型的值作为键名。) 实现
// 导出实现对象
export let Queue = (function(){
    // 声明空的WeakMap 来存储
  const items = new WeakMap();
  // 创建类
  class Que {
      // 利用构造器调用时 存储 当前调用者的队列
      // weakMap 有get()获取数据  set() 存储数据 很方便
    constructor(){
      items.set(this,[]);
    }
      // 进队
    enqueue(item){
      let q = items.get(this);
      q.push(item)
    }
      // 出队
    dequeue(){
      let q = items.get(this);
      return q.shift();
    }
      // 当前队列元素个数
    peek(){
      let q = items.get(this);
      return q.length > 0 ? q[q.length -1] : undefined;
    }
      // 当前队列个数
    size(){
      let q = items.get(this);
      return q.length;
    }
      // 清空当前队列
    clear(){
      let q = items.get(this);
       q = [];
    }
      // 当前队列是否为空
    isEmpty(){
      let q = items.get(this);
      return q.length <= 0; 
    }
      // 返回当前对列（返回数组）
    toArray(){
      return items.get(this);
    }
  }
  return Que;
})()

// 使用场景：直播时  某某进入直播间 
	// 动画显示 某某进入直播间需要时间，如果当同时进入多人时需要进入的人加入队列，以队列方式去展示动画

```



##### 2.栈：

```js
// 栈 在js中也是数组
// 栈的遵循规律是 先进后出，后进先出
// 实现队列会使用到的数组方法：
	//  pop() 出栈
	// push() 进栈
// 需要实现的基础栈的方法：
	// push(element(s)):添加一个（或几个）新元素到栈顶。
	// pop():移除栈顶的元素，同时返回被移除的元素。
	// peek():返回栈顶的元素，不对栈做任何的修改（这个方法不会移除栈顶的元素，仅仅是返回它）
	// isEmpty():如果栈里没有任何元素都返回true，否则返回false。
	// clear():移除栈里的所有元素
	// size():返回栈里的元素个数，这个方法和数组的length属性很类似。

class Stack{
    // 构造器中 声明一个空栈
  constructor(){
    this.items = [];
  }
    // 进栈
  push(item){
    this.items.push(item);
  }
    // 出栈
  pop(){
    return this.items.pop();
  }
    // 栈最上面的元素（最先出栈或者是最后进栈的元素）
  peek(){
    return this.items.length ? this.items[this.items.length - 1] : undefined;
  }
    // 栈是否为空
  isEmpty(){
    return items.length === 0;
  }
    // 清空栈
  clear(){
    this.items = [];
  }
    // 栈中元素的个数
  size(){
    return this.items.length;
  }
}





```
